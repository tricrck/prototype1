SQL Preview 

SELECT
 DATE_TRUNC('day', timestamp) AS "Date/Time",
 transport_mode AS "Transport Mode",
 route_location AS "Route/Location",
 SUM(traffic_volume) AS "Traffic Volume",
 AVG(revenue_kes) AS "Revenue (KES)"
FROM transport_data
WHERE timestamp BETWEEN '2024-01-01' AND '2024-03-15'
GROUP BY
 DATE_TRUNC('day', timestamp),
 transport_mode,
 route_location
ORDER BY "Date/Time" DESC;
